            <script>
        function fun(id, user_ans, right_ans){
            // Get all options for the current question
            let questionName = id.getElementsByTagName('input')[0].name;
            let options = document.getElementsByName(questionName);
            
            // Remove previous styling from all options of this question
            for (let i = 0; i < options.length; i++) {
                let optDiv = options[i].parentNode.parentNode;
                optDiv.style.backgroundColor = '';
                optDiv.style.color = 'black';
                optDiv.classList.remove('correct', 'incorrect');
            }
            
            // Apply new styling
            if(user_ans === right_ans){
                id.style.backgroundColor = 'green';
                id.style.color = 'white';
                id.classList.add('correct');
            } else {
                // Style user's wrong answer
                id.style.backgroundColor = 'red';
                id.style.color = 'white';
                id.classList.add('incorrect');

                // Highlight the correct answer
                // Iterate again to find the correct option div
                for (let i = 0; i < options.length; i++) {
                    if (options[i].value === right_ans) {
                        let correctOptDiv = options[i].parentNode.parentNode;
                        // Check if the correct option is not the one the user just selected (red)
                        if (correctOptDiv !== id) {
                           correctOptDiv.style.backgroundColor = 'darkgreen';
                           correctOptDiv.style.color = 'white';
                           correctOptDiv.classList.add('correct');
                        }
                        break; // Found the correct one, exit loop
                    }
                }
            }
        }
